import os
# 抑制 TensorFlow 的 INFO 和 WARNING 日志（可选）
os.environ['TF_CPP_MIN_LOG_LEVEL'] = '2'

import numpy as np
import matplotlib.pyplot as plt
import tensorflow as tf
from tensorflow.keras import layers, models
from tensorflow.keras.applications import ResNet50
from tensorflow.keras.preprocessing.image import ImageDataGenerator
from tensorflow.keras.callbacks import EarlyStopping, ReduceLROnPlateau
from sklearn.metrics import classification_report, confusion_matrix
from sklearn.utils.class_weight import compute_class_weight
from sklearn.model_selection import train_test_split
import datetime
from google.colab import drive
import pandas as pd
import warnings
warnings.filterwarnings('ignore')

# 2. Mount Google Drive
drive.mount('/content/drive')

# 3. Set paths
INPUT_PATH = '/content/drive/MyDrive/2026_lecture/Digital_health_pt/Midterm exam/Fracture/input'
OUTPUT_BASE = '/content/drive/MyDrive/2026_lecture/Digital_health_pt/Midterm exam/Fracture/output'
os.makedirs(OUTPUT_BASE, exist_ok=True)
timestamp = datetime.datetime.now().strftime('%Y%m%d_%H%M%S')
output_dir = os.path.join(OUTPUT_BASE, f'run_{timestamp}')
os.makedirs(output_dir, exist_ok=True)

print(f"Input folder: {INPUT_PATH}")
print(f"Output folder: {output_dir}")

# 4. Load image paths and labels
train_dir = os.path.join(INPUT_PATH, 'train')
if not os.path.exists(train_dir):
    raise FileNotFoundError(f"Training folder not found: {train_dir}")

fractured_dir = os.path.join(train_dir, 'fractured')
normal_dir = os.path.join(train_dir, 'normal')
if not os.path.exists(fractured_dir) or not os.path.exists(normal_dir):
    raise FileNotFoundError("Missing 'fractured' or 'normal' subfolder.")

filepaths = []
labels = []
for fname in os.listdir(fractured_dir):
    filepaths.append(os.path.join(fractured_dir, fname))
    labels.append(1)   # 1 = fractured
for fname in os.listdir(normal_dir):
    filepaths.append(os.path.join(normal_dir, fname))
    labels.append(0)   # 0 = normal

print(f"Total images: {len(filepaths)} (fractured: {sum(labels)}, normal: {len(labels)-sum(labels)})")

# 5. Stratified split (if total images >= 4; otherwise skip validation)
if len(filepaths) < 4:
    raise ValueError("Not enough images. Need at least 4 images (2 per class) for train/val split.")

X_train, X_val, y_train, y_val = train_test_split(
    filepaths, labels, test_size=0.2, random_state=42, stratify=labels
)

print(f"Training samples: {len(X_train)}, Validation samples: {len(X_val)}")

# 6. Data generators with stronger augmentation for small dataset
IMG_SIZE = (224, 224)
BATCH_SIZE = 4  # very small batch size

train_datagen = ImageDataGenerator(
    rescale=1./255,
    rotation_range=20,
    width_shift_range=0.2,
    height_shift_range=0.2,
    shear_range=0.2,
    zoom_range=0.2,
    horizontal_flip=True,
    fill_mode='nearest'
)

val_datagen = ImageDataGenerator(rescale=1./255)

def create_generator(file_list, label_list, datagen, batch_size, shuffle=True):
    df = pd.DataFrame({'filename': file_list, 'class': label_list})
    # Convert integer labels to strings (0->'normal', 1->'fractured')
    df['class'] = df['class'].map({0: 'normal', 1: 'fractured'})
    generator = datagen.flow_from_dataframe(
        df,
        x_col='filename', y_col='class',
        target_size=IMG_SIZE, batch_size=batch_size,
        class_mode='binary', shuffle=shuffle
    )
    return generator

train_gen = create_generator(X_train, y_train, train_datagen, BATCH_SIZE, shuffle=True)
val_gen = create_generator(X_val, y_val, val_datagen, BATCH_SIZE, shuffle=False)

# 7. Class weights
class_weights = compute_class_weight('balanced', classes=np.unique(y_train), y=y_train)
class_weight_dict = {0: class_weights[0], 1: class_weights[1]}
print(f"Class weights: {class_weight_dict}")

# 8. Build model (simpler head for tiny data)
print("\n=== Building model ===")
base_model = ResNet50(weights='imagenet', include_top=False, input_shape=(224,224,3))
base_model.trainable = False

model = models.Sequential([
    base_model,
    layers.GlobalAveragePooling2D(),
    layers.Dense(32, activation='relu'),   # smaller than before
    layers.Dropout(0.6),
    layers.Dense(1, activation='sigmoid')
])

model.compile(optimizer=tf.keras.optimizers.Adam(learning_rate=1e-4),
              loss='binary_crossentropy',
              metrics=['accuracy'])
model.summary()

# 9. Train phase 1
print("\n=== Training (phase 1) ===")
early_stop = EarlyStopping(monitor='val_loss', patience=5, restore_best_weights=True, verbose=1)
reduce_lr = ReduceLROnPlateau(monitor='val_loss', factor=0.5, patience=3, verbose=1)

history1 = model.fit(
    train_gen,
    epochs=30,
    validation_data=val_gen,
    class_weight=class_weight_dict,
    callbacks=[early_stop, reduce_lr],
    verbose=1
)

# 10. Fine-tuning
print("\n=== Fine-tuning (phase 2) ===")
base_model.trainable = True
# Freeze first 100 layers (keep only last few trainable)
for layer in base_model.layers[:100]:
    layer.trainable = False

model.compile(optimizer=tf.keras.optimizers.Adam(learning_rate=1e-5),
              loss='binary_crossentropy',
              metrics=['accuracy'])

history2 = model.fit(
    train_gen,
    epochs=15,
    validation_data=val_gen,
    class_weight=class_weight_dict,
    callbacks=[early_stop, reduce_lr],
    verbose=1
)

# 11. Evaluation
print("\n=== Evaluation ===")
val_loss, val_acc = model.evaluate(val_gen, verbose=0)
print(f"Validation Accuracy: {val_acc:.4f}")

# Predict
val_gen.reset()
predictions = model.predict(val_gen)
pred_classes = (predictions > 0.5).astype(int).flatten()

# Fix: val_gen.labels is a list, convert to numpy array
true_labels_list = val_gen.labels
true_classes = np.array(true_labels_list).astype(int)

class_names = ['normal', 'fractured']
print("\nClassification Report:")
print(classification_report(true_classes, pred_classes, target_names=class_names))

# Confusion matrix
cm = confusion_matrix(true_classes, pred_classes)
plt.figure(figsize=(5,4))
plt.imshow(cm, cmap='Blues')
plt.colorbar()
plt.xticks([0,1], class_names)
plt.yticks([0,1], class_names)
for i in range(2):
    for j in range(2):
        plt.text(j, i, cm[i,j], ha='center', va='center')
plt.title('Confusion Matrix')
plt.savefig(os.path.join(output_dir, 'confusion_matrix.png'))
plt.show()

# 12. Plot training curves
acc = history1.history['accuracy'] + history2.history['accuracy']
val_acc_hist = history1.history['val_accuracy'] + history2.history['val_accuracy']
loss = history1.history['loss'] + history2.history['loss']
val_loss_hist = history1.history['val_loss'] + history2.history['val_loss']

plt.figure(figsize=(12,4))
plt.subplot(1,2,1)
plt.plot(acc, label='Training Acc')
plt.plot(val_acc_hist, label='Validation Acc')
plt.title('Accuracy')
plt.legend()

plt.plot(loss, label='Training Loss')
plt.plot(val_loss_hist, label='Validation Loss')
plt.title('Loss')
plt.legend()
plt.tight_layout()
plt.savefig(os.path.join(output_dir, 'training_history.png'))
plt.show()

# 13. Sample predictions (only if validation set has images)
if len(X_val) > 0:
    sample_indices = np.random.choice(min(6, len(X_val)), min(6, len(X_val)), replace=False)
    fig, axes = plt.subplots(1, len(sample_indices), figsize=(3*len(sample_indices), 4))

        color = 'green' if true_label == pred_label else 'red'
        ax.imshow(img)

print(f"\nModel saved to: {output_dir}/fracture_model.keras")
